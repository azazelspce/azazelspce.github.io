              request_user  = GET( "https://api.github.com/user", gtoken )
              request_repos = GET( "https://api.github.com/user/repos?per_page=100", gtoken )

              process_request = function( x ) { fromJSON( toJSON( content( x ) ) ) }

              stop_for_status( request_user )
              stop_for_status( request_repos )

              user_data_frame  = process_request( request_user )
              repos_data_frame = process_request( request_repos )

              user_name = user_data_frame$login

              user_rows = grepl( user_name, repos_data_frame$full_name )
              repos_data_frame = repos_data_frame[user_rows, ]
              languages_url = repos_data_frame$languages_url

              result = lapply( languages_url, function( x ) process_request( GET( x, gtoken ) ) )
              valid_index = c( sapply( result, length ) != 0 )
              result = result[valid_index]
              repos_data_frame = repos_data_frame[valid_index, ]

              for( i in 1:length( result ) )
                result[[i]]$private = unlist( repos_data_frame[i, ]$private )

              all_names   = vector()
              names_count = vector()
              names_size  = vector()

              for( item in result )
              {
                if( !item$private )
                  for( name in names( item )[-length( item )] )
                  {
                    if( name %in% all_names )
                    {
                      names_size[name]  = names_size[name] + item[[name]]
                      names_count[name] = names_count[name] + 1
                    } else
                    {
                      all_names         = c( all_names, name )
                      names_size[name]  = item[[name]]
                      names_count[name] = 1
                    }
                  }
              }

              names_size  = names_size[order( names_size )]
              names_count = names_count[order( names_count )]

              base = "var data = [{\n__data__\n}];\n"

              AddNode = function( node_name, parent_node = "root", has_children = F )
              {
                children_str = "\"name\":\"__name__\",\n\"children\":\n[\n__children___name____\n]"

                if( parent_node == "root" )
                {
                  if( has_children )
                    result = gsub( "__data__", children_str, base )
                  else
                    result = gsub( "__data__", "\"name\":\"__name__\"", base )
                  return( gsub( "__name__", node_name, result ) )
                }
                else
                {
                  parent_str = "__children___name____"
                  parent_str = gsub( "__name__", parent_node, parent_str )

                  if( has_children )
                    children_str = paste( c( "{\n", children_str, "\n},\n", parent_str ), collapse = "" )
                  else
                    children_str = paste( c( "{\n\"name\":\"__name__\"\n},\n", parent_str ), collapse = "" )
                  result = gsub( parent_str, children_str, base )
                  return( gsub( "__name__", node_name, result ) )
                }
              }

              base = AddNode( node_name = "Search methods", has_children = T )

              base = AddNode( node_name = "Numerical methods"     , parent_node = "Search methods", has_children = T )
              base = AddNode( node_name = "Stochastic search"     , parent_node = "Search methods", has_children = T )
              base = AddNode( node_name = "Enumerative techniques", parent_node = "Search methods", has_children = F )

              base = AddNode( node_name = "Direct"  , parent_node = "Numerical methods", has_children = F )
              base = AddNode( node_name = "Indirect", parent_node = "Numerical methods", has_children = F )

              base = AddNode( node_name = "Single point"  , parent_node = "Stochastic search", has_children = F )
              base = AddNode( node_name = "Multiple point", parent_node = "Stochastic search", has_children = T )

              base = AddNode( node_name = "Swarm intelligence"     , parent_node = "Multiple point", has_children = F )
              base = AddNode( node_name = "Evolutionary strategies", parent_node = "Multiple point", has_children = T )

              base = AddNode( node_name = "Genetic algorithms", parent_node = "Evolutionary strategies", has_children = F )

              base = strsplit( base, "\n" )[[1]]
              base = gsub( "__.*__", "", base )
              base = paste( base, collapse = "" )
              base = gsub( ",]", "]", base )

              html_string = paste( readLines( "tree_diagram_fragment.txt" ), collapse = "\n" )

              html_string = gsub( "__json_data__", base               , html_string )
              html_string = gsub( "__div_name__" , "OptimizationChart", html_string )
              html_string = gsub( "__width__"    , 800               , html_string )
